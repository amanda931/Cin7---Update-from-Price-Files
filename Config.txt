UPDATE_SHOPIFY: False

# Exclude "Bathroom Brands" to avoid accidentally running 72k products

EXCLUDE_BATHROOM_BRANDS: False

# --- Filters ---

BRAND_FILTER: Tissino

#   overwrite  = the file always wins (file is the source of truth)
#   fill_blank = only set the attribute when Cin7's existing value is empty
ATTRIBUTE_FILL_MODE: overwrite

# --- Zapier safety prompt ---
# Before any LIVE run, prompt to confirm the sync Zap(s) are turned OFF so a bulk

ZAP_PAUSE_PROMPT: True
ZAP_NAME: Cin7 sync Zap(s)

# ==============================================================================
# --- Manufacturer uplift mode ---
# ==============================================================================
# When UPLIFT_MODE is True, the script does NOT read a price file. Instead it
# finds every product matching BRAND_FILTER / CATEGORY_FILTER above and lifts
# each one's existing Tier10 anchor by UPLIFT_PERCENT, recomputing all tiers,
# simPRO and Shopify prices from the new anchor.
#
# Safety: refuses to run if BOTH filters are blank; a live run shows a
# pre-flight summary (including the product count) and requires you to type
# CONFIRM, on top of the usual business-hours check.
#
# Always run with DRY_RUN: True first to preview the count and the changes.

UPLIFT_MODE: True
UPLIFT_GUARD: True
UPLIFT_DEFAULT_SUPPLIER: 

# Percentage to add (e.g. 5 = +5%)
UPLIFT_PERCENT: 7

# False = move selling prices only, hold the supplier (buy) cost until you have
#         the real price file  [recommended default]
# True  = also raise the supplier cost by the same %, treating the announced
#         increase as the true cost change
UPLIFT_SUPPLIER_COST: False

# Cin7 ignores Brand/Category filters on its product endpoint, so the whole
# catalogue is scanned once (~2-3 min) and cached to catalogue_index.json, then
# filtered locally. The cache rebuilds automatically once it is older than
# CATALOGUE_MAX_AGE_HOURS, so there is nothing to manage by hand day to day.
CATALOGUE_MAX_AGE_HOURS: 24

# Force a rebuild on the next run regardless of age (e.g. right after adding new
# products you want included straight away). No need to reset it — it only forces
# the next run.
REFRESH_CATALOGUE: False

# ==============================================================================
# --- Create missing products (Phase 2) ---
